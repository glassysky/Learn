include

int main() { int b, n; int mod, digit; int num[100] = {0}; int count;

scanf("%d", &b);
scanf("%d", &n);
while (b > 0) {
    digit = b / n;
    mod = b % n;
    b = digit;

    num[count++] = mod;

}
for (int i = 0; i < count; i++) {
    printf("%d", num[count - i]);
}
return 0;

}
